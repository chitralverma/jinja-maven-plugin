55  /**
56   * {@link GenerateFromJinjaTemplateMojo}
57   *
58   * <p>This mojo automatically renders concrete resources from Jinja template
59   * files as part of Maven build process.
60   *
61   * <p>Users define template file(s) and corresponding value file(s). When the
62   * plugin executes, it substitutes the values from value file(s) in the template
63   * file(s) and renders concrete resource(s) at the configured location.
64   *
65   * <p>This plugin uses jinjava, see <a
66   * href="https://github.com/HubSpot/jinjava#jinjava" target="_blank"> this
67   * link</a> for more info.
68   */
69  @Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_RESOURCES)
70  public class GenerateFromJinjaTemplateMojo extends AbstractMojo {
71  
72    /** Configuration to skip the entire goal. Default: false */
73    @Parameter(property = "jinja-maven.skip", defaultValue = "false")
74    private final Boolean skip = Boolean.FALSE;
75  
76    /** Configuration to fail if values for template are missing. Default: true */
77    @Parameter(
78        property = "jinja-maven.failOnMissingValues",
79        defaultValue = "true")
80    private final Boolean failOnMissingValues = Boolean.TRUE;
81  
82    /**
83     * Configuration to control if output files can be overwritten. Default: false
84     */
85    @Parameter(property = "jinja-maven.overwriteOutput", defaultValue = "false")
86    private final Boolean overwriteOutput = Boolean.FALSE;
87  
88    /**
89     * Configuration for resource set. A resource set is bundle of one or more
90     * resources which can be translated to a rendering job. Each resource
91     * contains a template file path, one or more value files and an output file
92     * path.
93     */
94    @Parameter(required = true)
95    private final List<ResourceBean> resourceSet = Collections.emptyList();
96  
97    /**
98     * Entry point to rendering logic. The whole process can be optionally skipped
99     * if required using the `Skip` configuration.
100    *
101    * <p>Step 1: Perform validation of configuration values provided by the user.
102    * Step 2: For each resource bundle, render the concrete files as per the
103    * provided template by substituting values from the value files. Step 3:
104    * Write concrete outputs as files.
105    *
106    * @throws MojoExecutionException Rendering errors result in
107    *     `MojoExecutionException`
108    * @throws MojoFailureException Validations errors result in
109    *     `MojoFailureException`
110    */
111   public void execute() throws MojoExecutionException, MojoFailureException {
112     getLog().debug("Plugin execution begins.");
113 
114     if (Boolean.TRUE.equals(skip)) {
115       getLog()
116           .warn(
117               String.format(
118                   "jinja:%s is skipped as %s=true", DEFAULT_PLUGIN_GOAL, SKIP));
119     } else {
120       validate();
121       printConfigs();
122 
123       getLog().info("Starting resource rendering process.");
124       for (ResourceBean resource : resourceSet) {
125         getLog().debug(String.format("Rendering resource '%s'", resource));
126 
127         String renderedResource = renderFromResource(resource);
128         writeOutput(resource.getOutputFilePath(), renderedResource);
129       }
130 
131       getLog().info("Resource rendering process is complete.");
132     }
133 
134     getLog().debug("Plugin execution ends.");
135   }
136 
137   /** Prints the configuration values provided by the user to debug level. */
138   private void printConfigs() {
139     ObjectMapper mapper = new ObjectMapper();
140     mapper.configure(SerializationFeature.INDENT_OUTPUT, true);
141 
142     try {
143       ObjectNode configuration = mapper.createObjectNode();
144       configuration.set(SKIP, BooleanNode.valueOf(skip));
145       configuration.set(RESOURCE_SET, new POJONode(resourceSet));
146       configuration.set(
147           FAIL_ON_MISSING_VALUES, new POJONode(failOnMissingValues));
148       configuration.set(OVERWRITE_OUTPUT, new POJONode(overwriteOutput));
149 
150       String jsonConfig = mapper.writeValueAsString(configuration);
151       getLog().debug(String.format("Plugin Config:\n%s", jsonConfig));
152     } catch (JsonProcessingException e) {
153       getLog().warn("Unable to print configs", e);
154     }
155   }

157   /**
158    * Validates the configuration values provided by the user.
159    *
160    * @throws MojoFailureException Validations errors result in
161    *     `MojoFailureException`
162    */
163   private void validate() throws MojoFailureException {
164     getLog().debug("Starting validations.");
165 
166     validateResourceSet();
167     getLog().debug("Validations complete");
168   }
169 
170   /**
171    * Validates the resource set values provided by the user.
172    *
173    * @throws MojoFailureException Validations errors result in
174    *     `MojoFailureException`
175    */
176   private void validateResourceSet() throws MojoFailureException {
177     if (resourceSet.isEmpty()) {
178       throw new MojoFailureException(
179           "Error occurred during configuration validation.",
180           new IllegalArgumentException(
181               String.format(
182                   "'%s' must be defined with at least 1 resource.",
183                   resourceSet)));
184     }
185 
186     for (ResourceBean resource : resourceSet) {
187       getLog().debug(String.format("Validating resource '%s'", resource));
188       validateResource(resource);
189     }
190   }
191 
192   /**
193    * Validates a resource of resource set as defined by the user.
194    *
195    * @param resource A user defined resource
196    * @throws MojoFailureException Validations errors result in
197    *     `MojoFailureException`
198    */
199   private void validateResource(ResourceBean resource)
200       throws MojoFailureException {
201     if (resource == null) {
202       throw new MojoFailureException(
203           "Error occurred during configuration validation.",
204           new IllegalArgumentException(
205               "Malformed 'resource' was encountered."));
206     }
207 
208     validateFile("templateFilePath", resource.getTemplateFilePath());
209 
210     if (resource.getValueFiles().isEmpty()) {
211       throw new MojoFailureException(
212           "Error occurred during configuration validation.",
213           new IllegalArgumentException(
214               "'valueFiles' must be defined with at least 1 path."));
215     }
216 
217     for (File file : resource.getValueFiles()) {
218       validateFile("valueFile", file);
219     }
220 
221     validateOutputFile(resource.getOutputFilePath());
222     validateDependencies(resource.getDependencyDirs());
223   }
224 
225   /**
226    * Validates a file based on path provided by the user for a resource.
227    *
228    * @param key Identifier for the type of file path (templateFile or valueFile)
229    * @param file Representation of file and directory path
230    * @throws MojoFailureException Validations errors result in
231    *     `MojoFailureException`
232    */
233   private void validateFile(String key, File file) throws MojoFailureException {
234     if (file == null) {
235       throw new MojoFailureException(
236           "Error occurred during configuration validation.",
237           new IllegalArgumentException(
238               String.format("'%s' path must not be null.", key)));
239     }
240 
241     if (!file.exists()) {
242       throw new MojoFailureException(
243           "Error occurred during configuration validation.",
244           new IllegalArgumentException(
245               String.format(
246                   "Provided %s at location '%s' does not exist.", key, file)));
247     }
248 
249     if (!file.isFile()) {
250       throw new MojoFailureException(
251           "Error occurred during configuration validation.",
252           new IllegalArgumentException(
253               String.format(
254                   "Provided %s at location '%s' must be a file.", key, file)));
255     }
256   }
257 
258   /**
259    * Validates rules are different for output files.
260    *
261    * @param file Representation of file and directory path
262    * @throws MojoFailureException Validations errors result in
263    *     `MojoFailureException`
264    */
265   private void validateOutputFile(File file) throws MojoFailureException {
266     if (file == null) {
267       throw new MojoFailureException(
268           "Error occurred during configuration validation.",
269           new IllegalArgumentException(
270               "'outputFilePath' path must not be null."));
271     }
272 
273     if (file.exists()) {
274       if (file.isDirectory()) {
275         throw new MojoFailureException(
276             "Error occurred during configuration validation.",
277             new IllegalArgumentException(
278                 String.format(
279                     "'outputFilePath' path '%s' must be a file.", file)));
280       } else if (!overwriteOutput) {
281         throw new MojoFailureException(
282             "Error occurred during configuration validation.",
283             new IllegalArgumentException(
284                 "Overwriting output files has been disabled in plugin config."
285                     + " Set 'overwriteOutput' config to true to allow this."));
286       }
287 
288       getLog()
289           .warn(
290               String.format(
291                   "'outputFilePath' path '%s' already exists "
292                       + "and will be overwritten.",
293                   file));
294     }
295   }

297   /**
298    * Validates path provided as one or more dependencies location.
299    *
300    * @param dirs Directories to be added to FileLocator.
301    * @throws MojoFailureException Validations errors result in
302    *     `MojoFailureException`
303    */
304   private void validateDependencies(List<File> dirs)
305       throws MojoFailureException {
306     if (dirs == null || dirs.isEmpty()) {
307       getLog().debug("No dependencies defined.");
308     } else {
309       for (File file : dirs) {
310         if (file == null) {
311           throw new MojoFailureException(
312               "Error occurred during configuration validation.",
313               new IllegalArgumentException(
314                   "'dependencyPath' path must not be null."));
315         }
316 
317         if (!file.exists()) {
318           throw new MojoFailureException(
319               "Error occurred during configuration validation.",
320               new IllegalArgumentException(
321                   String.format(
322                       "Provided dependencyPath at location '%s' does not exist.",
323                       file)));
324         }
325 
326         if (file.isFile()) {
327           throw new MojoFailureException(
328               "Error occurred during configuration validation.",
329               new IllegalArgumentException(
330                   String.format(
331                       "Provided dependencyPath at location '%s' must be a directory.",
332                       file)));
333         }
334       }
335     }
336   }

338   /**
339    * Rendering logic using Jinjava.
340    *
341    * <p>Value file(s) are read as JSON Objects using jackson and all the nodes
342    * are iterated add keys and typed values to a common context which will hold
343    * all values for substitution into the template.
344    *
345    * <p>Reason to choose JSON format for value files: - Type safety of values -
346    * Unstructured - Support complex types - Human readable and popular
347    *
348    * <p>Once the rendering is complete, errors are thrown if required.
349    *
350    * @param resource A user defined resource
351    * @return Rendered content as string
352    * @throws MojoExecutionException Rendering errors result in
353    *     `MojoFailureException`
354    */
355   private String renderFromResource(ResourceBean resource)
356       throws MojoExecutionException {
357     JinjavaConfig jc =
358         JinjavaConfig.newBuilder()
359             .withFailOnUnknownTokens(failOnMissingValues)
360             .build();
361     Jinjava jinjava = new Jinjava(jc);
362     addDependencyLocators(jinjava, resource.getDependencyDirs());
363 
364     Map<String, Object> context = Maps.newHashMap();
365 
366     try {
367       String templateContent =
368           FileUtils.readFileToString(
369               resource.getTemplateFilePath(), StandardCharsets.UTF_8);
370 
371       for (File valueFile : resource.getValueFiles()) {
372         ObjectMapper mapper = new ObjectMapper();
373         Iterator<Map.Entry<String, JsonNode>> iter =
374             mapper.readTree(valueFile).fields();
375 
376         while (iter.hasNext()) {
377           Map.Entry<String, JsonNode> next = iter.next();
378           JsonNodeType nodeType = next.getValue().getNodeType();
379 
380           if (next.getKey().contains(".")) {
381             throw new MojoExecutionException(
382                 "Error occurred during resource rendering.",
383                 new IllegalArgumentException(
384                     "Keys of value files cannot contain chars in [.]"));
385           }
386 
387           getLog().debug(String.format("Added entry [ %s ] to context.", next));
388           if (nodeType == JsonNodeType.ARRAY
389               || nodeType == JsonNodeType.OBJECT) {
390             context.put(next.getKey(), next.getValue());
391           } else {
392             context.put(next.getKey(), next.getValue().asText());
393           }
394         }
395       }
396 
397       RenderResult renderResult =
398           jinjava.renderForResult(templateContent, context);
399 
400       if (!renderResult.getErrors().isEmpty() && failOnMissingValues) {
401         throw new MojoExecutionException(
402             "Error occurred during resource rendering.",
403             new IllegalArgumentException(
404                 renderResult.getErrors().stream()
405                     .map(Object::toString)
406                     .collect(Collectors.joining(","))));
407 
408       } else {
409         return renderResult.getOutput();
410       }
411     } catch (IOException e) {
412       throw new MojoExecutionException(
413           "Error occurred during resource rendering.", e);
414     }
415   }

417   /**
418    * Allows users locate external resource(s) like external templates to
419    * include/ extend/ import from local file system.
420    *
421    * @param jinjava Jinja context for Java
422    * @param dependencyDirs List of user provided dependency directories
423    * @throws MojoExecutionException Exceptions occurred while creation of
424    *     locators are wrapped as `MojoExecutionException`.
425    */
426   private void addDependencyLocators(Jinjava jinjava, List<File> dependencyDirs)
427       throws MojoExecutionException {
428     List<ResourceLocator> resourceLocatorsList = Lists.newArrayList();
429     resourceLocatorsList.add(new ClasspathResourceLocator());
430 
431     for (File dir : dependencyDirs) {
432       try {
433         resourceLocatorsList.add(new FileLocator(dir));
434       } catch (FileNotFoundException e) {
435         throw new MojoExecutionException(
436             "Error occurred while creating resource locator.", e);
437       }
438     }
439 
440     ResourceLocator[] resourceLocators =
441         resourceLocatorsList.toArray(new ResourceLocator[0]);
442     jinjava.setResourceLocator(new CascadingResourceLocator(resourceLocators));
443   }

445   /**
446    * Writes the rendered content to a file.
447    *
448    * @param outputFile Output file as defined in some resource.
449    * @param renderedContent Content to be written to file as output.
450    * @throws MojoExecutionException `IOException` are recorded if any.
451    */
452   private void writeOutput(File outputFile, String renderedContent)
453       throws MojoExecutionException {
454     try {
455       FileUtils.writeStringToFile(
456           outputFile, renderedContent, StandardCharsets.UTF_8, false);
457     } catch (IOException e) {
458       throw new MojoExecutionException(
459           "Error occurred while writing output.", e);
460     }
461   }
